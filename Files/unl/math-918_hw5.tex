\begin{enumerate}[itemsep=4em]

\item In this question we see some generally useful comm alg facts\footnote{Not included is the useful but harder to prove fact that $A$ is normal if and only if $A$ satisfies $(R_1)$ and $(S_2)$.}. 
In particular, we are NOT assuming char $p$! First, some definitions:
\begin{defn}
Let $A$ be a Noetherian ring and let $k\in \bb N$.
\begin{itemize}
\item $A$ satisfies \emph{Serre's condition $(R_k)$} if for all primes $\mf p$ of height $\le k$, we have $A_{\mf p}$ is a regular local ring.

\item $A$ satisfies \emph{Serre's condition $(S_k)$} if for all primes $\mf p$, we have $\depth(A_{\mf p})\ge \min\{k,\operatorname{ht}(\mf p)\}$.
\end{itemize}
\end{defn}
Now for the problem:
\begin{enumerate}

\item Give an example of a ring that has $(S_1)$ but is not reduced.

\item Give an example of a ring that has $(R_0)$ but is not reduced.



\item Prove that $A$ is reduced if and only if $A$ satisfies $(R_0)$ and $(S_1)$.



\item Prove that $A$ is Cohen-Macaulay if and only if $A$ satisfies $(S_d)$ for $d=\dim A$, if and only if  $A$ satisfies $(S_k)$ for all $k$.\footnote{This has nothing to do with anything else on this HW, but is such a cute fact that I'm including it anyways.}


\end{enumerate}

\item In this question, we are back to assuming $A$ has char $p$.
\begin{enumerate}
\item Let $(A,\mf m)$ be a local char $p>0$ ring of dimension 0. Prove that $A$ is F-injective if and only if it is reduced.


\item Let $(A,\mf m)$ be a local char $p>0$ ring of dimension $>0$. Prove that if $A$ is F-injective then $H^0_{\mf m}(A)=0$.

\item Let $A$ be an $F$-injective ring. Prove that $A$ is reduced.

\end{enumerate}

\item (Exercise 22 of [MP]) Let R be an $\bb N$-graded ring over a field of prime characteristic $p > 0$ with
homogeneous maximal ideal $\mf m$. Show that
\begin{enumerate}
\item If R is $F$-injective, then $[H^i_{\mf m}(R)]_{>0} = 0$ for each $i$. Conclude that $a(R)\le 0$.


\item If R is $F$-rational, then $[H^d_{\mf m}(R)]_{\ge 0} = 0$. Conclude that $a(R)<0$.

\end{enumerate}

\item Here are some more useful colon ideal facts:
\begin{enumerate}
\item
Let $I$ ideal of a regular local ring $(R,\mf m)$. Prove that
\[
\Ass(I^{[p^e]}:I) \subseteq \Ass(I).
\]
Conclude that if $I$ is $\mf p$-primary then $I^{[p^e]}:I$ is $\mf p$-primary.

\item 
Let $(R,\mf m)$ be a regular local ring and $I$ unmixed (i.e., all associated primes are minimal). Let $I=\bigcap_{i=1}^n Q_i$. Prove that 
\[
I^{[p]}:I = \bigcap_{i} (Q_i^{[p]}:Q_i).
\]


\end{enumerate}
